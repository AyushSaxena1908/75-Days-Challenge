class Solution {
  public:
    // Function to find length of longest increasing subsequence.
    int longestSubsequence(vector<int>& arr) {
        // code here
        int n = arr.size();
    vector<int> longestSubsequence(n, 1);

    for (int i = 1; i < n; i++) {
        for (int prev = 0; prev < i; prev++) {
            if (arr[i] > arr[prev] && longestSubsequence[i] < longestSubsequence[prev] + 1) {
                longestSubsequence[i] = longestSubsequence[prev] + 1;
            }
        }
    }
    return *max_element(longestSubsequence.begin(), longestSubsequence.end());
}
};
